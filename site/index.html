quickStart.ts 
// Install: npm install @clicks-protocol/sdk ethers@^6

import { ClicksClient } from '@clicks-protocol/sdk';
import { ethers } from 'ethers';

const provider = new ethers.JsonRpcProvider(
  'https://mainnet.base.org'
);
const signer = new ethers.Wallet(
  process.env.PRIVATE_KEY, provider
);

// One call does everything:
// register + approve + split + route to yield
const clicks = new ClicksClient(signer);
await clicks.quickStart({
  agent: agentAddress,
  amount: '1000'
});
// → 800 USDC liquid, 200 USDC earning yield
operations.ts 
// Receive a payment with auto-split
await clicks.receivePayment('500', agentAddr);
// → 400 USDC liquid, 100 USDC to yield

// Check yield status
const info = await clicks.getYieldInfo();
// → { activeProtocol: 'morpho',
//     aaveAPY: 7.2, morphoAPY: 9.1 }

// Withdraw everything
await clicks.withdrawYield(agentAddr);
// → principal + yield to agent wallet
// → 2% of yield to protocol

// Custom yield split (5-50%)
await clicks.setOperatorYieldPct(30);
// → now 70% liquid, 30% to yield

// Simulate before committing
const sim = await clicks.simulateSplit('1000', addr);
// → { liquid: 700, toYield: 300, yieldPct: 30 }
