SDK

Two files. That's it. Install, copy, deploy.

quickStart.ts
import { ClicksSDK } from '@clicks-protocol/sdk' const clicks = new ClicksSDK({ rpcUrl: process.env.BASE_RPC_URL, privateKey: process.env.AGENT_PRIVATE_KEY, }) // Register + deposit in one call const result = await clicks.quickStart({ amount: '1000', // 1000 USDC referrer: '0xABC...', // optional }) console.log(`Earning yield on $${result.deposited} USDC`)
operations.ts
// Deposit more USDC await clicks.deposit({ amount: '500' }) // Check current balance + yield const balance = await clicks.getBalance() console.log(balance.principal) // '1500.00' console.log(balance.yield) // '12.34' // Withdraw to agent wallet await clicks.withdraw({ amount: '200' }) // Update split ratio (70/30) await clicks.updateSplit({ yieldPercent: 30 }) // Claim referral rewards const rewards = await clicks.claimReferralRewards()
